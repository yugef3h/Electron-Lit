/**
 * 给定一个整数数组 nums ，找到一个具有最大和的连续子数组（子数组最少包含一个元素），返回其最大和
 * @param {*} nums
 */

```js
var maxSubArray = function(nums) {
  let ans = nums[0], sum = 0
  for (let i=1; i<nums.length; i++) {
    if (sum + nums[i] < nums[i]) {
      sum = nums[i]
    } else {
      sum = ans + nums[i]
    }
    ans = Math.max(sum, ans)
  }
  return ans
}
```
